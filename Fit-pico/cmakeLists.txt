# git submodule add https://github.com/raspberrypi/pico-sdk.git deps/pico-sdk

cmake_minimum_required(VERSION 3.13)

if (NOT PICO_SDK_PATH AND NOT DEFINED ENV{PICO_SDK_PATH})
set(PICO_SDK_PATH "${CMAKE_CURRENT_LIST_DIR}/../deps/pico-sdk" CACHE PATH "Path to the Raspberry Pi Pico SDK")
endif()

if (NOT PICOTOOL_FETCH_FROM_GIT_PATH AND NOT DEFINED ENV{PICOTOOL_FETCH_FROM_GIT_PATH})
set(_FITPICO_LOCAL_PICOTOOL_DIR "${CMAKE_CURRENT_LIST_DIR}/../build/sensor/_deps")
if (EXISTS "${_FITPICO_LOCAL_PICOTOOL_DIR}/picotool/picotool")
set(PICOTOOL_FETCH_FROM_GIT_PATH "${_FITPICO_LOCAL_PICOTOOL_DIR}" CACHE PATH "Shared picotool cache" FORCE)
else()
set(PICOTOOL_FETCH_FROM_GIT_PATH "${CMAKE_CURRENT_LIST_DIR}/../.pico-tools" CACHE PATH "Shared picotool cache" FORCE)
endif()
unset(_FITPICO_LOCAL_PICOTOOL_DIR)
endif()

set(PICO_BOARD pico2_w CACHE STRING "Board type")

include(pico_sdk_import.cmake)
project(my_project C CXX ASM)

pico_sdk_init()

set(WIFI_SSID "bindsoft" CACHE STRING "Wi-Fi SSID for Pico W examples")
set(WIFI_PASSWORD "bindsoft24" CACHE STRING "Wi-Fi password for Pico W examples")

if (DEFINED ENV{MQTT_SERVER} AND (NOT MQTT_SERVER))
    set(MQTT_SERVER $ENV{MQTT_SERVER})
    message("Using MQTT_SERVER from environment ('${MQTT_SERVER}')")
endif()
if (NOT MQTT_SERVER)
    set(MQTT_SERVER "163.152.213.101" CACHE STRING "MQTT broker host for Pico W examples")
endif()
set(MQTT_SERVER "${MQTT_SERVER}" CACHE STRING "MQTT broker host for Pico W examples" FORCE)

if (DEFINED ENV{MQTT_USERNAME} AND (NOT MQTT_USERNAME))
    set(MQTT_USERNAME $ENV{MQTT_USERNAME})
    message("Using MQTT_USERNAME from environment ('${MQTT_USERNAME}')")
endif()
set(MQTT_USERNAME "${MQTT_USERNAME}" CACHE INTERNAL "MQTT user name for examples")

if (DEFINED ENV{MQTT_PASSWORD} AND (NOT MQTT_PASSWORD))
    set(MQTT_PASSWORD $ENV{MQTT_PASSWORD})
    message("Using MQTT_PASSWORD from environment")
endif()
set(MQTT_PASSWORD "${MQTT_PASSWORD}" CACHE INTERNAL "MQTT password for examples")

if (NOT MQTT_CERT_PATH)
    set(MQTT_CERT_PATH ${CMAKE_CURRENT_LIST_DIR}/src/mqtt/certs/${MQTT_SERVER})
endif()

if (NOT MQTT_CERT_INC)
    set(MQTT_CERT_INC mqtt_client.inc)
endif()


# ── 팀원 A: 센서 노드 ──────────────────────────────────────────────────────
# 센싱 + MQTT publish 전용
add_executable(fitpico_sensor
    src/main_sensor.c
)

target_link_libraries(fitpico_sensor
    pico_stdlib
    hardware_gpio
    hardware_spi
    pico_cyw43_arch_lwip_threadsafe_background
    pico_lwip_mqtt
)

target_include_directories(fitpico_sensor PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/common
    ${CMAKE_CURRENT_LIST_DIR}/src
)

target_compile_definitions(fitpico_sensor PRIVATE
    WIFI_SSID=\"${WIFI_SSID}\"
    WIFI_PASSWORD=\"${WIFI_PASSWORD}\"
    MQTT_SERVER=\"${MQTT_SERVER}\"
    MQTT_USERNAME=\"${MQTT_USERNAME}\"
    MQTT_PASSWORD=\"${MQTT_PASSWORD}\"
    PICO_STDIO_USB_CONNECT_WAIT_TIMEOUT_MS=3000
)

pico_enable_stdio_usb(fitpico_sensor 1)
pico_enable_stdio_uart(fitpico_sensor 0)
pico_add_extra_outputs(fitpico_sensor)

# ── 팀원 B: 디스플레이 노드 (I2C LCD + Buzzer) ────────────────────────────
